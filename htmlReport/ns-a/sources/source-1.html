 package dev.funkos.controllers;
 
 import dev.funkos.dto.FunkoCreateDto;
 import dev.funkos.dto.FunkoResponseDto;
 import dev.funkos.dto.FunkoUpdateDto;
 import dev.funkos.models.Funko;
 import dev.funkos.services.FunkoServiceImpl;
 import dev.storage.services.StorageService;
 import jakarta.validation.Valid;
 import lombok.NonNull;
 import lombok.extern.slf4j.Slf4j;
 import org.springframework.beans.factory.annotation.Autowired;
 import org.springframework.http.HttpStatus;
 import org.springframework.http.MediaType;
 import org.springframework.http.ResponseEntity;
 import org.springframework.validation.FieldError;
 import org.springframework.web.bind.MethodArgumentNotValidException;
 import org.springframework.web.bind.annotation.*;
 import org.springframework.web.multipart.MultipartFile;
 import org.springframework.web.server.ResponseStatusException;
 
 import java.util.HashMap;
 import java.util.List;
 import java.util.Map;
 
 @Slf4j
 @RequestMapping("/api/funkos")
 @RestController
 public class FunkoController {
 
     private final FunkoServiceImpl funkoService;
     private final StorageService storageService;
 
     @Autowired
     public FunkoController(FunkoServiceImpl funkoService, StorageService storageService) {
         this.funkoService = funkoService;
         this.storageService = storageService;
     }
 
     @GetMapping("/{id}")
     public Funko getFunkoById(@PathVariable Long id) {
         log.info("Buscando funko con ID: " + id);
         return funkoService.findById(id);
     }
 
     @GetMapping()
     public ResponseEntity<List<Funko>> getFunkos(@RequestParam(required = false) String categoria, @RequestParam(required = false) String nombre) {
         log.info("Buscando funkos...");
         return ResponseEntity.ok(funkoService.findAll(categoria, nombre));
     }
 
     @NonNull
     @PostMapping()
     public ResponseEntity<Funko> createFunko(@Valid @RequestBody FunkoCreateDto funko) {
         log.info("Creando funko...");
         return ResponseEntity.status(HttpStatus.CREATED).body(funkoService.save(funko));
     }
 
     @NonNull
     @PutMapping("/{id}")
     public ResponseEntity<Funko> updateFunko(@PathVariable Long id, @Valid @RequestBody FunkoUpdateDto funko) {
         log.info("Actualizando funko con ID: " + id);
         return ResponseEntity.ok(funkoService.update(funko, id));
     }
 
     @NonNull
     @PatchMapping("/{id}")
     public ResponseEntity<Funko> funkoPatch(@PathVariable Long id, @Valid @RequestBody FunkoUpdateDto funko) {
         log.info("Actualizando funko con ID: " + id);
         return ResponseEntity.ok(funkoService.update(funko, id));
     }
 
 
     @DeleteMapping("/{id}")
     public ResponseEntity<Void> deleteFunko(@PathVariable Long id) {
         funkoService.deleteById(id);
         return ResponseEntity.status(HttpStatus.NO_CONTENT).build();
     }
 
     @ResponseStatus(HttpStatus.BAD_REQUEST)
     @ExceptionHandler(MethodArgumentNotValidException.class)
     public Map<String, String> handleValidationExceptions(
             MethodArgumentNotValidException ex) {
         Map<String, String> errors = new HashMap<>();
         ex.getBindingResult().getAllErrors().forEach(error -> {
             String fieldName = ((FieldError) error).getField();
             String errorMessage = error.getDefaultMessage();
             errors.put(fieldName, errorMessage);
         });
         return errors;
     }
 
     @PatchMapping(value = "/image/{id}", consumes = MediaType.MULTIPART_FORM_DATA_VALUE)
     public ResponseEntity<FunkoResponseDto> uploadImage(@PathVariable Long id, @RequestPart("file") MultipartFile
             file) {
         log.info("Actualizando imagen de funko con id: " + id);
         if (!file.isEmpty()) {
             return ResponseEntity.ok(funkoService.updateImage(id, file, true));
 
         } else {
             throw new ResponseStatusException(HttpStatus.BAD_REQUEST, "No se ha enviado una imagen para el funko o esta está vacía");
         }
     }
 }
